;;; Emacs key bindings

;; after open Emacs, to load the recents files
	Space f r

;; create a file
	C-x f
	
;; save a file
	C-x s

;; check the active buffers
;; you can navigate and watch them with the cursor
;; it's not necessary to hit enter, only lay down in the one you need
;; and press escape to exit the minibuffer
	C-x b

;; kill the current buffer
	C-x k

;; close the current buffer
	C-x 0
	
;; expand the current buffer
	C-x 1

;; split the current buffer horizontally
	C-x 2

;; split the current buffer vertically
	C-x 3
